Consider the polynomial $p(x) = x^5 + x^2 - x + 1$. Newton’s method applied to
the polynomial $p$ is given by
\begin{align*}
&x_{0} \quad \quad \quad \textrm{initial seed}\\
&x_{n+1} = x_{n} - \frac{p(x_{n})}{p'(x_{n})}
\end{align*}

\begin{enumerate}[label=(\alph*)]
    \item Write the expression of the Newton’s method applied to the polynomial $p$.
    \begin{align*}
    &p(x) = x^5 + x^2 - x + 1\\
    &p'(x) = 5x^4 + 2x - 1
    \end{align*}
    \begin{align*}
    &x_{0} \quad \quad \quad \textrm{initial seed}\\
    &x_{n+1} = x_{n} - \frac{x_{n}^5 + x_{n}^2 - x_{n} + 1}{5x_{n}^4 + 2x_{n} - 1}
    \end{align*}

where:

\begin{itemize}
    \item $x_{n}$ is the $n$th approximation of the root of $p(x)$.
    \item $x_{n+1}$ is the $(n+1)$th approximation of the root of $p(x)$.
    \item $p(x_{n})$ is the value of $p(x)$ evaluated at $x_{n}$.
    \item $p'(x_{n})$ is the first derivative of $p(x)$ evaluated at $x_{n}$.
\end{itemize}

\pagebreak
